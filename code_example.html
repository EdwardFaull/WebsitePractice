namespace reverseWord 
	{ 
		class Program 
		{ 
			static void Main(string[] args) 
			{ 
				string word = ""; 
				Console.Write("Enter a word: "); 
				word = Console.ReadLine(); 
				for (int x = 0; x < word.Length; x++) 
				{ 
					Console.Write(word[word.Length - 1 - x]); 
				} 
				Console.WriteLine(); 
				Console.ReadLine(); 
			} 
		} 
	}
